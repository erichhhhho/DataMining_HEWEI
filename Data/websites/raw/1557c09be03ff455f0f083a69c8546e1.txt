 function EncodeUtf8(cn)
{
	var cnc = escape(cn);
	var sa = cnc.split("%");
	var retV ="";
	if(sa[0] != "")
	{
		retV = sa[0];
	}
	for(var i = 1; i < sa.length; i ++)
	{
		if(sa[i].substring(0,1) == "u")
		{
			retV += Hex2Utf8(Str2Hex(sa[i].substring(1,5)))+sa[i].substring(5,sa[i].length);
		}
		else retV += "%" + sa[i];
	}
	return retV;
}


function Str2Hex(s)
{
	var c = "";
	var n;
	var ss = "0123456789ABCDEF";
	var digS = "";
	for(var i = 0; i < s.length; i ++)
	{
		c = s.charAt(i);
		n = ss.indexOf(c);
		digS += Dec2Dig(eval(n));
	}
	return digS;
}

function Dec2Dig(n1)
{
	var s = "";
	var n2 = 0;
	for(var i = 0; i < 4; i++)
	{
		n2 = Math.pow(2,3 - i);
		if(n1 >= n2)
		{
			s += '1';
			n1 = n1 - n2;
		}
		else s += '0';
	}
	return s;
}

function Dig2Dec(s)
{
	var retV = 0;
	if(s.length == 4)
	{
		for(var i = 0; i < 4; i ++)
		{
			retV += eval(s.charAt(i)) * Math.pow(2, 3 - i);
		}
		return retV;
	}
	return -1;
}

function Hex2Utf8(s)
{
	var retS = "";
	var tempS = "";
	var ss = "";
	if(s.length == 16)
	{
		tempS = "1110" + s.substring(0, 4);
		tempS += "10" + s.substring(4, 10);
		tempS += "10" + s.substring(10,16);
		var sss = "0123456789ABCDEF";
		for(var i = 0; i < 3; i ++)
		{
			retS += "%";
			ss = tempS.substring(i * 8, (eval(i)+1)*8);
			retS += sss.charAt(Dig2Dec(ss.substring(0,4)));
			retS += sss.charAt(Dig2Dec(ss.substring(4,8)));
		}
		return retS;
	}
	return "";
}

<h1>
Trump congratulates Xi on being re-elected general secretary of CPC Central Committee
</h1>
<div class="info">
<span class="source">
Source: Xinhua
</span>
2017-10-26 02:19:06
</div>
</div>
<div id="content" class="mobileCon">
<div class="onlyVideo" id="mobileVideo"><iframe frameborder="0" style="width:100%; height:220px;" width="100%" height="220" scrolling="no" marginwidth="0" marginheight="0" src=""></iframe></div>
<p style="text-align: center; margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><img src="CnybnyE005006_20171025_NYMFN0A001_11n.jpg" style="height:630px; width:900px" id="{1161CAAE-43F9-496D-8F00-6C7439B96BD2}" /></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="color:#808080"><span style="font-family:arial,helvetica,sans-serif"><span style="font-size:16px">File Photo: Chinese President Xi Jinping (R) and his U.S. counterpart Donald Trump (L) hold the second round of talks in the Mar-a-Lago resort in Florida, the United States, April 7, 2017. (Xinhua/Lan Hongguang)</span></span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">BEIJING, Oct. 25 (Xinhua) -- U.S. President Donald Trump on Wednesday congratulated Chinese President Xi Jinping on being re-elected general secretary of the Central Committee of the Communist Party of China (CPC).</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Trump extended his congratulation in a phone conversation with Xi.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">He also congratulated Xi for the successful conclusion of the 19th CPC National Congress.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Trump said the 19th CPC National Congress has attracted worldwide attention, and he himself also paid close attention to important policy-related messages sent out by President Xi at the congress.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">On his upcoming state visit to China, Trump said the American people are heatedly discussing the visit, and he expects to meet President Xi in Beijing to fully exchange views on enhancing cooperation between the two countries and on international and regional issues of common concern.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Xi thanked Trump for the congratulation.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Xi said the just-concluded 19th CPC National Congress, which has drawn a blueprint for China's future development, is of great significance.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Xi said, as being noted in his report to the congress, China will unswervingly follow the path of peaceful development, pursue a win-win strategy of opening-up, expand converging interests with all countries, and promote coordination and cooperation among major </span></span><span style="font-family:georgia,serif; font-size:16px">world </span><span style="font-family:georgia,serif"><span style="font-size:16px">powers.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Xi stressed that China attaches great importance to the Sino-U.S. relations and is willing to promote the long-term, healthy and stable development of bilateral relations on the basis of mutual respect and mutual benefits.</span></span></p>
<p style="margin-top: 10px; margin-bottom: 10px; font-size: 16px; font-family: Georgia, serif;"><span style="font-family:georgia,serif"><span style="font-size:16px">Noting President Trump's upcoming visit to China, Xi said he expects to plan with Trump in Beijing the future development for Sino-U.S. relations so as to promote cooperation between the two sides, bring real benefits to the two peoples and make greater contributions to regional and international peace, stability and prosperity.</span></span></p>
